## 深度克隆

```ts
function deepClone(obj, hash = new WeakMap()) {
  // 处理基本类型和函数
  if (obj === null || typeof obj !== 'object') {
    if (typeof obj === 'symbol') return Symbol(obj.description);
    return obj; // 包括 number, string, boolean, undefined, bigint, function
  }

  // 处理循环引用
  if (hash.has(obj)) return hash.get(obj);

  // 获取对象类型
  const type = Object.prototype.toString.call(obj);
  
  // 处理特殊对象类型
  switch (type) {
    case '[object Date]':
      return new Date(obj.getTime());
      
    case '[object RegExp]': {
      const clone = new RegExp(obj.source, obj.flags);
      clone.lastIndex = obj.lastIndex;
      return clone;
    }
      
    case '[object Map]': {
      const clone = new Map();
      hash.set(obj, clone);
      obj.forEach((value, key) => {
        clone.set(deepClone(key, hash), deepClone(value, hash));
      });
      return clone;
    }
      
    case '[object Set]': {
      const clone = new Set();
      hash.set(obj, clone);
      obj.forEach(value => {
        clone.add(deepClone(value, hash));
      });
      return clone;
    }
      
    case '[object ArrayBuffer]':
      return obj.slice();
      
    case '[object Array]':
    case '[object Arguments]': {
      const clone = type === '[object Arguments]' ? Array.from(obj) : [];
      hash.set(obj, clone);
      for (let i = 0; i < obj.length; i++) {
        clone[i] = deepClone(obj[i], hash);
      }
      return clone;
    }
      
    default:
      // 处理 TypedArray 和 DataView
      if (ArrayBuffer.isView(obj)) {
        return new obj.constructor(
          deepClone(obj.buffer, hash),
          obj.byteOffset,
          obj.length
        );
      }
      
      // 处理普通对象
      const clone = Object.create(Object.getPrototypeOf(obj));
      hash.set(obj, clone);
      
      // 使用 for-in 加 hasOwnProperty 检查，性能优于 Reflect.ownKeys
      for (const key in obj) {
        if (Object.prototype.hasOwnProperty.call(obj, key)) {
          clone[key] = deepClone(obj[key], hash);
        }
      }
      
      // 单独处理 Symbol 属性
      const symbolKeys = Object.getOwnPropertySymbols(obj);
      for (const symKey of symbolKeys) {
        clone[symKey] = deepClone(obj[symKey], hash);
      }
      
      return clone;
  }
}
```
